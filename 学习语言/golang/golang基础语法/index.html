4. 定义变量
第一种 [var identifier type] # 指定声明变量的类型
第二种 [var d = true] # 根据值自行判断变量的类型
第三种 [intVal := 123] # 声明&赋值

    当一个函数有3个返回值时，你只想使用两个，则可以 [ _, a, b = main() ]
当仅指定变量类型，没有初始化时，则变量值为默认值。
常见默认值：
    数值类型（包括complex64/128）为 0

    字符串为 ""（空字符串）

    var a []int
    var a map[string] int
    var a chan int
    var a func(string) int

7. 可见性[常量、变量、类型、函数名、结构字段等等]
以一个大写字母开头，可以被外部包的代码所使用；，如：Group1，那么使用这种形式的标识符的对象就可以被外部包的代码所使用（客户端程序需要先导入这个包），这被称为导出（像面向对象语言中的 public）
以小写字母开头，则对包外是不可见的；如：group1，但是他们在整个包的内部是可见并且可用的（像面向对象语言中的 protected ）。

		fmt.Printf("1==2")

		fmt.Printf("==default")
	}

func Test数据类型_Channel(t *testing.T) {
	// 定义两个通道
	c1 := make(chan string)
	c2 := make(chan string)
	// 向通道写入数据
	go func() { time.Sleep(1 * time.Second); fmt.Printf("数据写入通道c1"); c1 <- "one" }()
	go func() { time.Sleep(2 * time.Second); fmt.Printf("数据写入通道c2"); c2 <- "two" }()
	// 取出通道数据；使用 select 语句非阻塞地从两个通道中获取数据
	for i := 0; i < 2; i++ {
		// 当满足任意一个case时，才结束本次select；
		// 若每个case都不满足，则走default
		select {
		case msg1 := <-c1:
			fmt.Printf("channel 1 received %s", msg1)
		case msg2 := <-c2:
			fmt.Printf("channel 2 received %s", msg2)
		default:
			// 若没有default，则将一直循环判断每一个case
			fmt.Printf("default output")
		}
	}
}

	/* 声明map变量 <string,string> */
	var mapStringString map[string]string
	mapStringString = make(map[string]string)
	mapStringString["key"] = "value"

	/* 声明map变量 <string,string> */
	var mapStringMapIntString map[string]map[int]string     // 声明
	mapStringMapIntString = make(map[string]map[int]string) // 初始化
	mapStringMapIntString["key"] = make(map[int]string)     // 初始化
	mapStringMapIntString["key"][0] = "key-0-value"
	printInfo("<string,<int,string>> : %+v ", mapStringMapIntString)

	/* 创建 map + 初始化 */
	mapStringStringOneStep := make(map[string]string)
	printInfo("%+v ", mapStringStringOneStep)
}

func Test数据类型_指针(t *testing.T) {
	consoleColor()

	// 数据类型 - 指针
	// 指针的基本定义与使用
	printInfo("[1;40;32m 指针的基本定义与使用 [0m")
	ipStr := "192.168.1.1" // 1. 定义一个字符串
	var ip *string         // 2. 定义一个指针
	ip = &ipStr            // 3. 将指针ip 指向字符串的地址
	// 打印指针指向的地址
	printInfo("指针地址 %p", ip)
	// 打印指针指向地址的数据
	printInfo("指针数据 %+v", *ip)

	ageInt := 12
	var age *int
	age = &ageInt
	printInfo("%p", age)
	printInfo("%+v", *age)

	// 指向指针的指针
	printInfo("[1;40;32m 测试指向指针的指针 [0m")
	// 变量声明格式
	var ptr *int
	ptr = &ageInt
	var pptr **int
	pptr = &ptr
	printInfo("%p", &pptr)
	printInfo("%+v", pptr)
	// 指针pptr的值，存储的是指针ptr的地址，加*则打印该地址的值，也就是 &ageInt
	printInfo("%+v", *pptr)
	// ageInt
	printInfo("%+v", **pptr)

	// 向函数传递指针参数
	printInfo("[1;40;32m 测试向函数传递指针参数 [0m")
	var x = 1
	var y = 0
	printInfo("交换前x的值 %d", x)
	printInfo("交换前y的值 %d", y)
	swap(&x, &y)
	printInfo("交换后x的值 %d", x)
	printInfo("交换后y的值 %d", y)

}

func consoleColor() {
	fmt.Printf("")
	for b := 40; b <= 47; b++ { // 背景色彩 = 40-47
		for f := 30; f <= 37; f++ { // 前景色彩 = 30-37
			for d := range []int{0, 1, 4, 5, 7, 8} { // 显示方式 = 0,1,4,5,7,8
				fmt.Printf(" %c[%d;%d;%dm%s   (f=%d,b=%d,d=%d)   %c[0m ",
					0x1B, d, b, f, "", f, b, d, 0x1B)
			}
			fmt.Printf("")
		}
		fmt.Printf("")
	}
}
