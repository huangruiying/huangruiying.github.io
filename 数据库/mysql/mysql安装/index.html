MySQL安装

博客 标签 类目 主页

安装

brew install mysql@5.7 https://downloads.mysql.com/archives/community/

启动

sudo /usr/local/mysql-5.6.10-osx10.7-x86_64/bin/mysqld –user=mysql

初始登陆

mysql -u root -p

重置密码 root用户密码重置为root123

set password for root@localhost = password(‘root123’);

创建数据库 colorful

create database colorful charset=utf8 ;

进入数据库 colorful

use colorful;

在库 colorful 内创建一张表 c_test

CREATE TABLE c_test ( id bigint(20) NOT NULL AUTO_INCREMENT COMMENT ‘主键’, name varchar(100) NOT NULL COMMENT ‘姓名’, age int DEFAULT 1 COMMENT ‘年龄’, PRIMARY KEY (id), KEY idx_name (name) ) ENGINE=InnoDB AUTO_INCREMENT=30 DEFAULT CHARSET=utf8 COMMENT=‘测试表’

测试库连接

jdbc:mysql://127.0.0.1:3306/colorful?user=root&password=root123&useUnicode=true&characterEncoding=utf-8&useSSL=false

下载

cd /usr/local/mysql/bin/

获取权限

sudo su

禁止验证密码

./mysqld_safe –skip-grant-tables &

进入数据库重置密码

./mysql FLUSH PRIVILEGES; SET PASSWORD FOR ‘root’@’localhost’ = PASSWORD(‘root’);
