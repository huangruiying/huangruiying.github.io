CREATE TABLE `biz_order` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    `biz_order_no` VARCHAR(64) NOT NULL COMMENT '业务方订单号，幂等键',
    `biz_line` VARCHAR(64) NOT NULL COMMENT '业务线标识',
    `biz_identity` VARCHAR(64) DEFAULT NULL COMMENT '业务标识',
    `user_id` VARCHAR(64) NOT NULL COMMENT '内部用户ID',
    `total_amount` BIGINT NOT NULL COMMENT '订单总金额（分）',
    `currency` VARCHAR(16) DEFAULT 'CNY' COMMENT '币种',
    `order_status` VARCHAR(32) DEFAULT 'INIT' COMMENT '业务订单状态（INIT、PROCESSING、SUCCESS、CLOSED 等）',
    `notify_url` VARCHAR(255) DEFAULT NULL COMMENT '业务方通知地址',
    `sign_url` VARCHAR(255) DEFAULT NULL COMMENT '签约结果通知地址',
    `return_url` VARCHAR(255) DEFAULT NULL COMMENT '页面跳转地址',
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    PRIMARY KEY (`id`),
    UNIQUE KEY `uniq_biz_order_no` (`biz_order_no`),
    INDEX `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='业务收单单表（业务方发起的支付请求）';

支付核心单 pay_order ​

字段分类字段名说明
订单标识pay_order_no支付核心单号（支付内部生成）
biz_order_no业务方单号（接入业务方的核心单号，一个业务方单号对多个支付核心单号）
channel_order_no支付通道单号（第三方支付平台生成）
商户信息mch_no商户号（支付通道的商户号）
isv_no服务商号
mch_type商户类型：1-普通商户，2-特约商户（服务商模式）
金额相关pay_amount支付金额（单位：分）
refund_amount累计退款总金额（单位：分）
platform_subsidy自由平台补贴（单位：分）
biz_subsidy业务方补贴（单位：分）
channel_subsidy支付通道补贴（单位：分）
bank_subsidy银行补贴（单位：分）
支付状态pay_status支付状态：待支付、支付成功、超时关单、主动关单等
refund_status退款状态：0-未退款，1-部分退款，2-全额退款
refund_times退款次数
callback_status向下游回调状态：0-未发送，1-已发送
请求与结果request_data请求参数（支付服务下单时的原始参数 JSON）
result_data支付结果（通道/渠道回调、查询等 JSON）
支付方式与产品pay_type支付方式（卡基：储蓄卡、信用卡；账基：支付宝、微信、云闪付），收银台列表展示的是这些支付方式
trade_type支付产品：App支付、H5支付、小程序、扫码、付款码、信用付、代扣、周期支付
receipt_channel收单通道：京东、易宝、微信、云闪付、招商银行等
示例：使用易宝间连微信 → 易宝是收单通道，微信是支付方式
业务信息biz_line业务线标识
biz_identity业务标识
通知地址notify_url业务结果通知地址（如 http://xxxxxws）
return_url页面跳转地址（如 http://xxxxxws）
sign_notify_url签约结果通知地址（用于支付即签约场景）
用户信息user_id内部用户 ID
third_identity_id支付通道用户标识（如 openid、uid）
client_ip客户端 IP
shell
CREATE TABLE `pay_order` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    
    -- 订单标识
    `pay_order_no` VARCHAR(64) NOT NULL COMMENT '支付核心单号（支付内部生成）',
    `biz_order_no` VARCHAR(64) NOT NULL COMMENT '业务方单号（业务方的核心单号）',
    `channel_order_no` VARCHAR(64) DEFAULT NULL COMMENT '支付通道单号（第三方平台生成）',

    -- 商户信息
    `mch_no` VARCHAR(64) NOT NULL COMMENT '商户号（支付通道的商户号）',
    `isv_no` VARCHAR(64) DEFAULT NULL COMMENT '服务商号',
    `mch_type` TINYINT DEFAULT 1 COMMENT '商户类型：1-普通商户，2-特约商户（服务商模式）',

    -- 金额相关
    `pay_amount` BIGINT DEFAULT 0 COMMENT '支付金额（单位：分）',
    `refund_amount` BIGINT DEFAULT 0 COMMENT '累计退款金额（单位：分）',
    `platform_subsidy` BIGINT DEFAULT 0 COMMENT '平台补贴（单位：分）',
    `biz_subsidy` BIGINT DEFAULT 0 COMMENT '业务方补贴（单位：分）',
    `channel_subsidy` BIGINT DEFAULT 0 COMMENT '通道补贴（单位：分）',
    `bank_subsidy` BIGINT DEFAULT 0 COMMENT '银行补贴（单位：分）',

    -- 支付状态
    `pay_status` VARCHAR(32) DEFAULT NULL COMMENT '支付状态：待支付、支付成功、超时关单、主动关单等',
    `refund_status` TINYINT DEFAULT 0 COMMENT '退款状态：0-未退款，1-部分退款，2-全额退款',
    `refund_times` INT DEFAULT 0 COMMENT '退款次数',
    `callback_status` TINYINT DEFAULT 0 COMMENT '向下游回调状态：0-未发送，1-已发送',

    -- 请求与结果
    `request_data` TEXT COMMENT '请求参数（下单时的原始JSON）',
    `result_data` TEXT COMMENT '支付结果（通道回调/主动查询结果JSON）',

    -- 支付方式与产品
    `pay_type` VARCHAR(32) DEFAULT NULL COMMENT '支付方式（如 储蓄卡、信用卡、支付宝、微信、云闪付）',
    `trade_type` VARCHAR(32) DEFAULT NULL COMMENT '支付产品（App支付、H5、小程序、扫码等）',
    `receipt_channel` VARCHAR(32) DEFAULT NULL COMMENT '收单通道（如 京东、易宝、微信等）',

    -- 业务信息
    `biz_line` VARCHAR(64) DEFAULT NULL COMMENT '业务线标识',
    `biz_identity` VARCHAR(64) DEFAULT NULL COMMENT '业务标识',

    -- 通知地址
    `notify_url` VARCHAR(255) DEFAULT NULL COMMENT '支付结果通知地址',
    `return_url` VARCHAR(255) DEFAULT NULL COMMENT '页面跳转地址',
    `sign_notify_url` VARCHAR(255) DEFAULT NULL COMMENT '签约结果通知地址',

    -- 用户信息
    `user_id` VARCHAR(64) DEFAULT NULL COMMENT '内部用户ID',
    `third_identity_id` VARCHAR(128) DEFAULT NULL COMMENT '支付通道用户身份标识',
    `client_ip` VARCHAR(45) DEFAULT NULL COMMENT '客户端 IP',

    -- 元数据
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    PRIMARY KEY (`id`),
    UNIQUE KEY `uniq_pay_order_no` (`pay_order_no`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='支付订单主表';

id
场景编码：varchar
场景名称：描述这是用作什么场景收单的规则
场景状态：int 下线，上线
匹配策略：varchar 类名，入参为mapParam，出参为true/false。策略内会限制是全量mapParam做匹配，还是部分mapParam做匹配。
    宽松匹配，满足条件即可，即参数有 A B C D，对于场景规则 A B，A C D，A B C D 均可匹配成功。
    严格匹配，严格按照策略内规定好的参数，进行匹配。
规则脚本：varchar script_code

CREATE TABLE `rule_route` (
`id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键 ID',
`scene_code` VARCHAR(64) NOT NULL COMMENT '场景编码',

`provider_merchant_no` VARCHAR(64) DEFAULT NULL COMMENT '服务商商户号',
`merchant_no` VARCHAR(64) DEFAULT NULL COMMENT '商户号',
`channel_weight` INT DEFAULT 0 COMMENT '通道权重',
`channel_status` TINYINT DEFAULT 1 COMMENT '通道状态（0-下线，1-上线，2-暂停使用）',

`remark` VARCHAR(255) DEFAULT NULL COMMENT '备注',

支付与业务方的对接模式有哪些 ​

对接渠道的模式
渠道收款 - 外借商户号/渠道预付费
    支付结果由渠道通知，也可主动查询
    退款由渠道/客服系统发起
    退款驳回

自收款，渠道微信小程序
    收单类 ChWechatMiniPayService，返回appid + mini跳转参数(path,callback=channelPayParam)，前端跳转微信小程序。 (pages/webview/thirdpay/index?paySn=支付单号&path=payAggregation/channelPayParam&pft=10066001&orderNo=订单号)
    收银台 无，默认微信小程序
    通道下单 /channelPayParam，直连间连均返回TradeNo这种参数。

自收款，渠道App+H5收银台(无登录状态) 骑士卡api、快电、中移上研流程
    收单类 QishikaApiPayService；返回跳转收银台的连接
    收银台 h5PayInfoForBizChannel；依赖渠道号，查开放平台支付方式，根据映射关系找到我方PayTypeInfoConfig实际支付方式
    通道下单 h5ObtainPayInfoForBizChannel；二代路由。
        直连：返回标准参数
        间连：
            App 微信返回跳转小程序参数(pages/webview/payapp/index?paySn=支付单号&path=payAggregation/obtainWeChatMiniPayInfo&pft=10066001&orderNo=订单号) appWeChatMiniPaymentService
                支付宝返回scheme zfbIndirectPaymentService
            H5  微信返回scheme，包装obtainWeChatMiniPayInfo接口 h5WeChatMiniPaymentService 
                支付宝返回scheme，包装obtainZfbMiniPayParams接口 zfbIndirectPaymentService
            Mini/MP
                支付宝/微信都返回appid、paySign，前端分别在微信/支付宝里打开对应的小程序。 AdaPaymentService YeePaymentService
        
        返回scheme的还需再一次请求通道下单 obtainWeChatMiniPayInfo / obtainZfbMiniPayParams
