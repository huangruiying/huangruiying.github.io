实现对请求的不同响应结果，进行限流。（单机实验） ​

对接口A的请求参数、响应结果、异常情况进行不同频次的限流。 比如对同一个接口：

  • 调用成功的，默认频率限制为150QPS
  • 调用失败抛异常时，限流频率限制为0.1QPS
  • 调用成功但被校验拦截时，限流频率限制为2QPS
  • 针对请求参数中业务线为02的，限流20QPS
  • 针对响应结果包含某些标识内容的，限流5QPS

先简单做一个单机版的。
我的想法是通过自定义限流注解，控制在不同场景下，要限流多少QPS，持续多少时间。到期后再恢复默认的QPS。
当然，会存在多次触发同一限流场景(比如失败异常)，我采取的策略是「限流处于生效状态下时，立即应用当前失败场景的qps，但恢复默认qps不会向后延期。」

自定义限流注解 ​

java
package org.huangry.colorful.ratelimit;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

/**
 * 根据响应结果
 */
@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.METHOD)
public @interface RateLimitByResult {

	// 定义一个与其他限流操作隔离的标识
	String owner();

	// 限流标识：从方法参数提取
	// #{参数名称} + _ + #{参数名称.子参数.子参数}
	String key() default "";

	/**
	 * @return 默认限流频率
	 */
	double defaultLimit() default 10;

	/**
	 * @return 执行代码出现异常时，限流频率
	 */
	double exceptionLimit() default 1;

	/**
	 * 代码抛出异常时，限流 {@linkplain this#exceptionLimit()} 持续的时间。
	 * 当该时间超时后，限流将自动调整为默认的 {@linkplain this#defaultLimit()} 频率
	 */
	long exceptionMills() default 1000L;

	RateLimit[] config() default {};

	/**
	 * 此限流配置会强制控制流控参数
	 * 当一个时间窗口内，多个业务结果的condition均被匹配，则选择最小的limit进行流控
	 * <p>
	 * 强制要求：RateLimit内配置的限流值要求要比{@linkplain RateLimitByResult#defaultLimit}小
	 */
	@Retention(RetentionPolicy.RUNTIME)
	@interface RateLimit {

		/**
		 * 针对响应结果进行区分限流，SpEL表达式，表达式执行结果「强制要求」为Boolean类型。
		 * 表达式内对「响应对象」的标识为 #response
		 *
		 * @return SpEL
		 */
		String condition();

		// 限流值是多少
		double limit();

		// 限流持续多久
		long mills() default 1000L;
	}
}

定义限流切面处理逻辑 ​

java
package org.huangry.colorful.ratelimit;

import lombok.extern.slf4j.Slf4j;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.reflect.MethodSignature;
import org.huangry.colorful.ratelimit.limiter.LocalRateLimiterService;
import org.springframework.core.DefaultParameterNameDiscoverer;
import org.springframework.expression.ExpressionParser;
import org.springframework.expression.spel.standard.SpelExpressionParser;
import org.springframework.expression.spel.support.StandardEvaluationContext;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.lang.reflect.Method;
import java.util.Arrays;

/**
 * 动态调整限流器限流频率
 */
@Aspect
@Component
@Slf4j
public class RateLimitAspect {

	@Resource
	private LocalRateLimiterService localRateLimiterService;

	private final DefaultParameterNameDiscoverer nameDiscoverer = new DefaultParameterNameDiscoverer();

	@Around("@annotation(rateLimitAnno)")
	public Object limit(ProceedingJoinPoint joinPoint, RateLimitByResult rateLimitAnno) throws Throwable {

		// 当前方法的限流key
		String limitKey = getRateLimitKey(joinPoint, rateLimitAnno);

		try {
			// 限流控制
			rateLimit(limitKey, rateLimitAnno.defaultLimit());
		} catch (RateLimitException e) {
			// 命中限流，抛异常
			throw e;
		} catch (Exception e) {
			// 非限流的其他异常，忽略，不影响主流程
			log.error("rateLimit occur exception", e);
		}

		Object result;
		try {
			// 业务处理 & 业务结果
			result = joinPoint.proceed();
			if (result == null) {
				return null;
			}
		} catch (Exception e) {
			// 业务执行出现异常，调整限流参数
			localRateLimiterService.forceOverrideRate(limitKey,
					rateLimitAnno.exceptionLimit(),
					rateLimitAnno.defaultLimit(),
					rateLimitAnno.exceptionMills());
			// 并抛异常
			throw e;
		}

		try {
			// 动态限流：通过业务结果，匹配限流条件，并动态调整限流速率
			dynamicRateLimit(rateLimitAnno, result, limitKey);
		} catch (Exception e) {
			log.error("dynamicRateLimit occur exception", e);
			// 忽略动态限流的异常
		}
		return result;
	}

	/**
	 * 匹配限流条件，并动态调整限流速率
	 *
	 * @param rateLimitAnno 条件
	 * @param result        业务结果
	 * @param limitKey      限流key
	 */
	private void dynamicRateLimit(RateLimitByResult rateLimitAnno, Object result, String limitKey) {

		// 业务结果写入Context
		ExpressionParser parser = new SpelExpressionParser();
		StandardEvaluationContext context = new StandardEvaluationContext();
		context.setVariable("response", result);

		// 进行逐一匹配condition
		Arrays.stream(rateLimitAnno.config()).forEach(rateLimit -> {
			String cEL = rateLimit.condition().trim();

			// 用响应结果，匹配注解限流条件
			if (Boolean.TRUE.equals(parser.parseExpression(cEL).getValue(context, Boolean.class))) { // 满足限流条件时

				// 调整限流速率,持续xxx时间
				localRateLimiterService.forceOverrideRate(limitKey,
						rateLimit.limit(),
						rateLimitAnno.defaultLimit(),
						rateLimit.mills());
			}
		});
	}


	/**
	 * 依赖「当前切入方法」和「限流注解」确定限流key前缀
	 * 注意：当方法没有参数时，直接取{@link RateLimitByResult#owner()}作为限流key
	 *
	 * @param joinPoint     jp
	 * @param rateLimitAnno rma
	 * @return {@link RateLimitByResult#owner()} + "_" +  {@link RateLimitByResult#key()}
	 */
	private String getRateLimitKey(ProceedingJoinPoint joinPoint, RateLimitByResult rateLimitAnno) {
		try {
			// 1. 将方法的入参，按照[name,value]方式写入EvaluationContext
			String[] parameterNames = getParameterNames(joinPoint); // 参数名称
			Object[] args = joinPoint.getArgs(); // 参数值
			if (parameterNames == null || args == null || parameterNames.length == 0 || args.length == 0) {
				return rateLimitAnno.owner();
			}
			// 创建 SpEL 解析器
			ExpressionParser parser = new SpelExpressionParser();
			// 创建 EvaluationContext 并设置变量
			StandardEvaluationContext context = new StandardEvaluationContext();
			for (int i = 0; i < parameterNames.length; i++) {
				context.setVariable(parameterNames[i], args[i]);
			}

			// 2. 提取注解内SpEL表达式，并执行，获得限流拦截keyPrefix
			String owner = rateLimitAnno.owner();
			String keyExpr = rateLimitAnno.key();
			String value = parser.parseExpression(keyExpr).getValue(context, String.class);
			return owner + "_" + value;
		} catch (Exception e) {
			log.error("RateLimitAspect getRateLimitKey exception", e);
			return "";
		}
	}

	/**
	 * 获取方法参数名称
	 *
	 * @param joinPoint jp
	 * @return 参数名称数组
	 * 注意：当方法存在参数，但是返回的「参数名称数组」为空时
	 * 1.请确认maven是否开启「保留参数名」
	 * <p>
	 * <pre>
	 * {@code
	 *  <plugin>
	 *      <groupId>org.apache.maven.plugins</groupId>
	 *      <artifactId>maven-compiler-plugin</artifactId>
	 *      <configuration>
	 *          <parameters>true</parameters>
	 *      </configuration>
	 *  </plugin>
	 * }
	 * </pre>
	 * 2.请确认目标方法是否被代理，导致修改方法签名
	 */
	private String[] getParameterNames(ProceedingJoinPoint joinPoint) {
		// 获取方法签名
		MethodSignature signature = (MethodSignature) joinPoint.getSignature();
		Method method = signature.getMethod();
		// 获取方法参数名称
		return nameDiscoverer.getParameterNames(method);
	}

	/**
	 * 限流
	 *
	 * @param limitKey 当前命中限流的key
	 * @param rate     当前限流的阈值，当根据limitKey找不到限流器时，依赖此参数创建限流器。
	 */
	private void rateLimit(String limitKey, double rate) {
		// 限流
		if (! localRateLimiterService.tryAcquire(limitKey, rate)) {
			log.error("命中限流 {}", limitKey);
			throw new RateLimitException(401, "Rate limit exceeded for " + limitKey);
		}
	}

}

应用限流注解进行实验 ​

java
package org.huangry.colorful.demo.example.presentation.controller;

import lombok.extern.slf4j.Slf4j;
import org.huangry.colorful.demo.example.presentation.dto.ratelimit.RateLimitResp;
import org.huangry.colorful.ratelimit.aop.RateLimitByResult;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

@RestController
@RequestMapping("RateLimitByResultTest")
@Slf4j
public class RateLimitByResultTest {

	@RateLimitByResult(
			owner = "RateLimitByResultTest#executeSuccess",
			key = "#bizLine",
			config = {
					// 当业务异常时，限流1qps，持续10秒
					@RateLimitByResult.RateLimit(condition = "#response.code == 500", limit = 2, mills = 10 * 1000L),
					// 当响应指定数据 188 时，限流188qps，持续5秒
					@RateLimitByResult.RateLimit(condition = "#response.data.equals('188')", limit = 188, mills = 5 * 1000L),
			})
	@GetMapping("/executeSuccess")
	public RateLimitResp executeSuccess(@RequestParam("bizLine") String bizLine, @RequestParam("type") String type) {

		if ("EXCEPTION".equals(type)) {
			throw new RuntimeException();
		}

		try {
			if ("ERROR".equals(type)) {
				throw new RuntimeException("");
			}
			RateLimitResp rateLimitResp = new RateLimitResp();
			rateLimitResp.setCode(200);
			rateLimitResp.setMessage("成功executeSuccess");
			rateLimitResp.setData(type);
			log.info("【执行业务逻辑：成功case {} 】", bizLine);
			return rateLimitResp;
		} catch (Exception e) {
			RateLimitResp rateLimitResp = new RateLimitResp();
			rateLimitResp.setCode(500);
			rateLimitResp.setMessage("失败executeFailed");
			rateLimitResp.setData(null);
			log.info("【执行业务逻辑：异常case {} 】", bizLine);
			return rateLimitResp;
		}
	}
}
