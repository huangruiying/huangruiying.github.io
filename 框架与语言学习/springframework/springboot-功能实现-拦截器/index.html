本文旨在SpringBoot中如何创建一个拦截器。

/**
 * 定义拦截器
 * 
 * @author huangry
 * Created in 2023/7/21 10:18
 */
@Component
@Slf4j
public class SelfDefineInterceptor extends HandlerInterceptorAdapter {

	@Override
	public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
		log.info("beg SelfDefineInterceptor");
		return true;
	}

	@Override
	public void postHandle(HttpServletRequest request, HttpServletResponse response, Object handler, ModelAndView modelAndView) throws Exception {
		log.info("end SelfDefineInterceptor");
	}
}

@Configuration
public class SelfDefineWebMvcConfigurerAdapter extends WebMvcConfigurationSupport {// WebMvcConfigurer
	/**
	 * 注册拦截器
	 */
	@Autowired
	private SelfDefineInterceptor SelfDefineInterceptor;

	@Override
	public void addInterceptors(InterceptorRegistry registry) {
		// 注册拦截器,自定义不同拦截器拦截不同的地址
		registry.addInterceptor(new TestInterceptor1()).addPathPatterns("/ready");
		registry.addInterceptor(new TestInterceptor2()).addPathPatterns("/health");
	}
	// 若不配置拦截器addPathPattern地址，默认全拦截
    // 每个拦截器的addPathPatterns，excludePathPatterns添加的路径是各自独立的
}
