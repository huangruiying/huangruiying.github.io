SpEL表达式示例

On this page

SpEL 应用在@Value注解的几个示例 ​

支持运行时查询和操作对象，广泛应用于Spring体系，可以独立使用。

基础的字符串、数值类型注入 ​

java

@Value("${example.address: 默认地址}")
private String address;
@Value("${example.age: 18}")
private String age;

集合类注入: 简单结构 ​

java
// List
@Value("#{'${example.list.integer:11,12,13}'.split(',')}")
private List<Integer> integerList;
@Value("#{'${example.list.string:hello,omg,huangry}'.split(',')}")
private List<String> stringList;
// Map
@Value("#{${example.map.string.object:{'key_1':-1,'key_2':'2','key_3':'hallo hallo'}}}")
private Map<String, Object> objectMap;

集合类注入: map list嵌套 ​

java
// Map + List
@Value("#{${example.map.list.object:{-1:{1,2},1:{1},2:{2}}}}")
private Map<String, List<Integer>> listMap;

SpEL解析对象属性 ​

java
@Data
@AllArgsConstructor
class Person {
	private String name;
	private int age;
}

public class SpELObjectDemo {
    public static void main(String[] args) {
        // 1. 创建 SpEL 解析器
        ExpressionParser parser = new SpelExpressionParser();

        // 2. 创建 EvaluationContext 并设置变量
        StandardEvaluationContext context = new StandardEvaluationContext();
        Person person = new Person("John", 25);
        context.setVariable("p", person);

        // 3. 解析对象属性
        Expression exp = parser.parseExpression("#p.name");
        String name = exp.getValue(context, String.class);
        System.out.println("Person 名字: " + name); // 输出: John

        // 4. 解析方法调用
        Expression exp2 = parser.parseExpression("#p.getAge()");
        Integer age = exp2.getValue(context, Integer.class);
        System.out.println("Person 年龄: " + age); // 输出: 25
    }
}

SpEL表达式示例 ​

java
package org.huangry.colorful.demo.example.application.SpEL表达式;

import org.springframework.expression.ExpressionParser;
import org.springframework.expression.spel.standard.SpelExpressionParser;
import org.springframework.stereotype.Service;

/**
 * @author huangry
 * Created in 2024/1/3 12:50
 */
@Service
public class SpELBusiness {

	public static void main(String[] args) {
		ExpressionParser parser = new SpelExpressionParser();

		// 解析方法调用
		System.out.println(parser.parseExpression("'Hello World'.concat('!')").getValue());

		// 解析到 List<Integer>
		System.out.println(parser.parseExpression("{1,2,3,4}").getValue());

		// 解析到 List<List>
		System.out.println(parser.parseExpression("{{'a','b'},{'x','y'}}").getValue());

		// 解析到 Map<Integer,List<Integer>>
		System.out.println(parser.parseExpression("{-1:{1,2},1:{1},2:{2}}").getValue());

		// 解析到 Map<String,Objects>
		System.out.println(parser.parseExpression("{'key_1':-1,'key_2':'2','key_3':'hallo hallo'}").getValue());

		// 比较数字
		System.out.println(" 17 < 83 =   结果: " + parser.parseExpression(" 17 < 83 ").getValue());

		// 比较字符串
		System.out.println(" 'hello'.equals('hello') =   结果: " + parser.parseExpression(" 'hello'.equals('hello') ").getValue());

		// 数学运算
		System.out.println(" 6 + 6 =   结果: " + parser.parseExpression(" 6 + 6 ").getValue());
		
		// 三位运算
		System.out.println(" ( 5 > 4 ) ? hello : false   结果: " + parser.parseExpression(" ( 5 > 4 ) ? 'hello': 'false' ").getValue());
		
		// instanceof
		System.out.println(" instance示例  'a' instanceof T(String)   结果: " + parser.parseExpression("'a' instanceof T(String)").getValue());
		
		// 正则表达式
		System.out.println(" 78 是否为两位数字 :" + parser.parseExpression("78 matches '\\d{2}'").getValue());

	}
}
