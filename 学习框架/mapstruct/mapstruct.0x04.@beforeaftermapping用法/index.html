@Mapper
public interface BusinessConvertMapper {

	BusinessConvertMapper INSTANCE = Mappers.getMapper(BusinessConvertMapper.class);

	@Mappings(value = {
			@Mapping(target = "column1", ignore = true),
			@Mapping(target = "column2", ignore = true)}
	)
	ToModel convert(FromModel from);

	@BeforeMapping
	default void preProcess() {
		System.out.println("执行前置处理......");
	}

	/**
	 * 在调用转换方法后触发
	 * @param to 转换方法的响应类型
	 */
	@AfterMapping
	default void postProcess(@MappingTarget ToModel to) {
		// ToModel 的CouponName字段长度大于128时，做截断处理

			String substring = bill.getCouponName().substring(0, 128);
			bill.setCouponName(substring);
		}
	}
}
